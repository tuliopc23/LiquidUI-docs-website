## Quick Start ‚ö°

```bash
npm install liquidify
```

```jsx
import { GlassButton, GlassCard } from 'liquidify';

function App() {
  return (
    <GlassCard variant='glass'>
      <h1>Hello World</h1>
      <GlassButton variant='primary'>Get Started</GlassButton>
    </GlassCard>
  );
}
```
