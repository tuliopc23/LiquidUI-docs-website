# Design Token Visual Guide

Interactive visual representations of the Liquidify design token system.

## Spacing Token Visualization

### Spacing Scale

```tsx
import { liquidGlassTokens } from 'liquidify';

const SpacingVisualization = () => {
  const spacing = liquidGlassTokens.spacing;

  const spacingValues = [
    { key: '0', value: spacing[0], description: 'No spacing' },
    { key: '1', value: spacing[1], description: 'Hair line' },
    { key: '2', value: spacing[2], description: 'Tight' },
    { key: '3', value: spacing[3], description: 'Close' },
    { key: '4', value: spacing[4], description: 'Default' },
    { key: '5', value: spacing[5], description: 'Comfortable' },
    { key: '6', value: spacing[6], description: 'Loose' },
    { key: '8', value: spacing[8], description: 'Section' },
    { key: '10', value: spacing[10], description: 'Large' },
    { key: '12', value: spacing[12], description: 'Extra large' },
    { key: '16', value: spacing[16], description: 'Section divider' },
    { key: '20', value: spacing[20], description: 'Page section' },
    { key: '24', value: spacing[24], description: 'Major section' },
    { key: '32', value: spacing[32], description: 'Hero section' },

  return (
    <div className='space-y-4'>
```


## Interactive Token Explorer

```tsx
const TokenExplorer = () => {
  const [selectedCategory, setSelectedCategory] = useState('spacing');
  const [selectedToken, setSelectedToken] = useState(null);

  const categories = {
    spacing: liquidGlassTokens.spacing,
    colors: liquidGlassTokens.colors,
    typography: liquidGlassTokens.typography,
    animation: liquidGlassTokens.animation,
    glass: liquidGlassTokens.glass,
  };

  const renderTokenValue = (key, value) => {
    if (selectedCategory === 'colors' && typeof value === 'string') {
      return (
        <div className='flex items-center gap-2'>
          <div
            className='w-4 h-4 rounded border'
            style={{ backgroundColor: value }}
          />
          <span className='font-mono text-sm'>{value}</span>
        </div>
      );
    }

    if (selectedCategory === 'spacing' && typeof value === 'string') {
      return (
        <div className='flex items-center gap-2'>
          <div className='bg-blue-500 h-2' style={{ width: value }} />
          <span className='font-mono text-sm'>{value}</span>
        </div>
      );
    }

    return <span className='font-mono text-sm'>{JSON.stringify(value)}</span>;
  };

  return (
    <div className='border rounded-lg p-4'>
      <div className='flex gap-2 mb-4'>
        {Object.keys(categories).map(category => (
          <button
            key={category}
            onClick={() => setSelectedCategory(category)}
            className={`px-3 py-1 rounded text-sm ${
              selectedCategory === category
                ? 'bg-blue-500 text-white'
                : 'bg-gray-200 text-gray-700'
            }`}
          >
            {category}
          </button>
        ))}
      </div>

      <div className='space-y-2 max-h-96 overflow-y-auto'>
        {Object.entries(categories[selectedCategory]).map(([key, value]) => (
          <div
            key={key}
            className='flex justify-between items-center p-2 hover:bg-gray-50 rounded'
          >
            <span className='font-medium'>{key}</span>
            {renderTokenValue(key, value)}
          </div>
        ))}
      </div>
    </div>
  );
};
```
