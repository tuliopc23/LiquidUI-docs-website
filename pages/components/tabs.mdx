# Tabs

A beautiful glassmorphism tabs component for organizing content into separate panels that users can navigate between with smooth transitions and elegant visual design.

## TypeScript Interface

```typescript
interface GlassTabsProps {
  tabs: Array<{
    id: string;
    label: string;
    content: React.ReactNode;
    disabled?: boolean;
  }>;
  defaultTab?: string;
  className?: string;
  onTabChange?: (tabId: string) => void;
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `tabs` | `TabItem[]` | `[]` | Array of tab objects defining the tabs and their content |
| `defaultTab` | `string` | First tab ID | ID of the tab to be active by default |
| `className` | `string` | - | Additional CSS classes for custom styling |
| `onTabChange` | `(tabId: string) => void` | - | Callback function called when active tab changes |

### Tab Item Configuration

| Property | Type | Required | Description |
|----------|------|----------|-------------|
| `id` | `string` | ✓ | Unique identifier for the tab |
| `label` | `string` | ✓ | Display text for the tab button |
| `content` | `React.ReactNode` | ✓ | Content to display when tab is active |
| `disabled` | `boolean` | - | Whether the tab is disabled and non-clickable |

## Advanced Usage

### Dynamic Tab Loading

```tsx
import { GlassTabs } from '@tuliocunha23/liquidui';
import { useState, useEffect } from 'react';

interface TabData {
  id: string;
  label: string;
  content: React.ReactNode;
  disabled?: boolean;
}

function DynamicTabs() {
  const [tabs, setTabs] = useState<TabData[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadTabs();
  }, []);

  const loadTabs = async () => {
    setLoading(true);
    try {
      // Simulate API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      
      const dynamicTabs: TabData[] = [
        {
          id: "products",
          label: "Products",
          content: <ProductList />
        },
        {
          id: "analytics",
          label: "Analytics",
          content: <AnalyticsDashboard />
        },
        {
          id: "settings",
          label: "Settings",
          content: <SettingsPanel />,
          disabled: !userHasPermission('admin')
        }
      ];
      
      setTabs(dynamicTabs);
    } catch (error) {
      console.error('Failed to load tabs:', error);
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <div className="text-center py-8">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-500 mx-auto"></div>
        <p className="mt-2 text-gray-600">Loading tabs...</p>
      </div>
    );
  }

  return <GlassTabs tabs={tabs} />;
}

function ProductList() {
  return <div className="p-4">Product list content...</div>;
}

function AnalyticsDashboard() {
  return <div className="p-4">Analytics dashboard...</div>;
}

function SettingsPanel() {
  return <div className="p-4">Settings panel...</div>;
}
```

### Responsive Tab Behavior

```tsx
import { GlassTabs } from '@tuliocunha23/liquidui';
import { useMobile } from '@tuliocunha23/liquidui';

function ResponsiveTabs() {
  const isMobile = useMobile();

  const allTabs = [
    { id: "overview", label: "Overview", content: <div>Overview content</div> },
    { id: "details", label: "Details", content: <div>Details content</div> },
    { id: "specifications", label: "Specifications", content: <div>Specs content</div> },
    { id: "reviews", label: "Reviews", content: <div>Reviews content</div> },
    { id: "support", label: "Support", content: <div>Support content</div> },
  ];

  // Show fewer tabs on mobile
  const visibleTabs = isMobile ? allTabs.slice(0, 3) : allTabs;

  return (
    <div>
      <GlassTabs tabs={visibleTabs} />
      {isMobile && allTabs.length > 3 && (
        <div className="mt-4 text-center">
          <button className="text-blue-600 text-sm">
            View all {allTabs.length} tabs
          </button>
        </div>
      )}
    </div>
  );
}
```

### Form Wizard with Tabs

```tsx
import { GlassTabs } from '@tuliocunha23/liquidui';
import { useState } from 'react';

function FormWizard() {
  const [formData, setFormData] = useState({
    personal: {},
    contact: {},
    preferences: {}
  });
  
  const [completedSteps, setCompletedSteps] = useState<string[]>([]);

  const isStepComplete = (stepId: string) => completedSteps.includes(stepId);
  
  const tabs = [
    {
      id: "personal",
      label: `1. Personal ${isStepComplete('personal') ? '✓' : ''}`,
      content: (
        <PersonalInfoForm 
          data={formData.personal}
          onComplete={() => setCompletedSteps(prev => [...prev, 'personal'])}
        />
      )
    },
    {
      id: "contact",
      label: `2. Contact ${isStepComplete('contact') ? '✓' : ''}`,
      content: (
        <ContactInfoForm 
          data={formData.contact}
          onComplete={() => setCompletedSteps(prev => [...prev, 'contact'])}
        />
      ),
      disabled: !isStepComplete('personal')
    },
    {
      id: "preferences",
      label: `3. Preferences ${isStepComplete('preferences') ? '✓' : ''}`,
      content: (
        <PreferencesForm 
          data={formData.preferences}
          onComplete={() => setCompletedSteps(prev => [...prev, 'preferences'])}
        />
      ),
      disabled: !isStepComplete('contact')
    }
  ];

  return (
    <div>
      <h2 className="text-xl font-bold mb-4">Account Setup Wizard</h2>
      <GlassTabs tabs={tabs} />
    </div>
  );
}
```

### Tabs with URL Synchronization

```tsx
import { GlassTabs } from '@tuliocunha23/liquidui';
import { useRouter } from 'next/router';
import { useEffect, useState } from 'react';

function URLSyncedTabs() {
  const router = useRouter();
  const [defaultTab, setDefaultTab] = useState<string>('');

  const tabs = [
    { id: "general", label: "General", content: <div>General settings</div> },
    { id: "security", label: "Security", content: <div>Security settings</div> },
    { id: "billing", label: "Billing", content: <div>Billing settings</div> },
  ];

  useEffect(() => {
    // Set default tab from URL query parameter
    const tabFromUrl = router.query.tab as string;
    const validTab = tabs.find(tab => tab.id === tabFromUrl);
    setDefaultTab(validTab ? tabFromUrl : 'general');
  }, [router.query.tab]);

  const handleTabChange = (tabId: string) => {
    // Update URL without page reload
    router.push({
      pathname: router.pathname,
      query: { ...router.query, tab: tabId }
    }, undefined, { shallow: true });
  };

  if (!defaultTab) {
    return <div>Loading...</div>;
  }

  return (
    <GlassTabs
      tabs={tabs}
      defaultTab={defaultTab}
      onTabChange={handleTabChange}
    />
  );
}
```

## Accessibility

The GlassTabs component includes basic accessibility features:

- **Button Semantics**: Tab headers use proper button elements
- **Disabled States**: Properly handles disabled tabs with visual feedback
- **Keyboard Navigation**: Inherent keyboard support through button elements

## Styling

The GlassTabs component uses glassmorphism design with these default styles:

- **Container**: `backdrop-blur-md bg-white/10 border border-white/20 rounded-2xl`
- **Tab Buttons**: `rounded-xl px-6 py-3` with `transition-all duration-300`
- **Active Tab**: Enhanced glass effect with `bg-white/20` and shadow
- **Content Area**: Minimum height of 200px with glass styling
- **Disabled Tabs**: Reduced opacity (`opacity-50`) and disabled cursor

## Limitations

The current implementation has some limitations:

1. **No Controlled Mode**: Missing `activeId` prop for external state control
2. **No Style Variants**: Only one visual style available
3. **Limited ARIA Support**: Basic accessibility implementation

### Manual Controlled Implementation

```tsx
import { GlassTabs } from '@tuliocunha23/liquidui';
import { useState } from 'react';

function ControlledTabsWrapper() {
  const [activeTab, setActiveTab] = useState('tab1');

  // You would need to implement this manually since the component
  // doesn't support controlled mode natively
  return (
    <div>
      <div className="mb-4">
        <button onClick={() => setActiveTab('tab1')}>External Tab 1</button>
        <button onClick={() => setActiveTab('tab2')}>External Tab 2</button>
      </div>
      
      <GlassTabs
        key={activeTab} // Force re-render when activeTab changes
        tabs={tabs}
        defaultTab={activeTab}
        onTabChange={setActiveTab}
      />
    </div>
  );
}
```
