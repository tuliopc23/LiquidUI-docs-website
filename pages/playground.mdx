## Performance Testing

<CodePlayground
  title="Animation Performance"
  description="Test complex animations and performance optimizations"
  code={`function PerformanceTest() {
  const [itemCount, setItemCount] = useState(10);
  
  const items = Array.from({ length: itemCount }, (_, i) => ({
    id: i,
    title: \`Item \${i + 1}\`,
    description: \`Description for item \${i + 1}\`
  }));

  return (
    <div className="space-y-4">
      <div className="flex items-center gap-4">
        <label>Items: {itemCount}</label>
        <input
          type="range"
          min="1"
          max="100"
          value={itemCount}
          onChange={(e) => setItemCount(parseInt(e.target.value))}
          className="flex-1"
        />
      </div>
      
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {items.map((item) => (
          <div
            key={item.id}
            className="glass-card p-4 hover:scale-105 transition-transform"
          >
            <h3 className="font-semibold">{item.title}</h3>
            <p className="text-sm opacity-70">{item.description}</p>
          </div>
        ))}
      </div>
    </div>
  );
}`}
  preview={
    <ComponentShowcase>
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {Array.from({ length: 6 }, (_, i) => (
          <div
            key={i}
            className="backdrop-blur-xl bg-white/10 border border-white/20 rounded-xl p-4 hover:scale-105 transition-transform cursor-pointer"
          >
            <h3 className="font-semibold">Item {i + 1}</h3>
            <p className="text-sm opacity-70">Description for item {i + 1}</p>
          </div>
        ))}
      </div>
    </ComponentShowcase>
  }
/>
