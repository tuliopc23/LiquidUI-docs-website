    ```tsx
    import { useState } from 'react'
    import { Card, Input, Button } from '@tuliocunha23/liquidui'
    
    function LoginForm() {
      const [email, setEmail] = useState('')
      const [password, setPassword] = useState('')
      const [loading, setLoading] = useState(false)
      
      const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault()
        setLoading(true)
        
        // Simulate API call
        await new Promise(resolve => setTimeout(resolve, 2000))
        
        setLoading(false)
      }
      
      return (
        <div className="min-h-screen bg-gradient-to-br from-purple-400 via-pink-500 to-red-500 flex items-center justify-center p-4">
          <Card className="glass-effect max-w-md w-full">
            <div className="text-center mb-8">
              <h2 className="text-2xl font-bold text-white mb-2">Welcome Back</h2>
              <p className="text-white/80">Sign in to your account</p>
            </div>
            
            <form onSubmit={handleSubmit} className="space-y-6">
              <Input
                type="email"
                value={email}
                onChange={(e) => setEmail(e.target.value)}
                placeholder="Enter your email"
                className="glass-input"
                required
              />
              
              <Input
                type="password"
                value={password}
                onChange={(e) => setPassword(e.target.value)}
                placeholder="Enter your password"
                className="glass-input"
                required
              />
              
              <Button
                type="submit"
                variant="glass"
                fullWidth
                loading={loading}
                size="lg"
              >
                {loading ? 'Signing in...' : 'Sign In'}
              </Button>
            </form>
          </Card>
        </div>
      )
    }
    ```
  </Tab>
</Tabs>

## Best Practices

When building with LiquidiUI, follow these best practices:

### 1. Layer Management
- Use glass effects sparingly to maintain visual hierarchy
- Ensure sufficient contrast between glass layers
- Test readability across different backgrounds

### 2. Performance Optimization
- Limit the number of backdrop-filter elements on a single page
- Use CSS containment for better performance
- Consider fallbacks for older browsers

### 3. Accessibility
- Maintain proper color contrast ratios
- Ensure focus indicators are visible on glass elements
- Test with screen readers and keyboard navigation

### 4. Responsive Design
- Adjust glass effects for mobile devices
- Consider reducing blur intensity on smaller screens
- Test touch interactions on glass elements
