## Performance Considerations

### CSS Variable Optimization

- Use HSL color space for better interpolation
- Minimize the number of custom properties
- Group related variables together
- Use fallback values for better browser support

### Best Practices

1. **Consistent Naming**: Use semantic names for variables
2. **Fallback Values**: Always provide fallbacks for custom properties
3. **Performance**: Avoid excessive nesting of glass effects
4. **Accessibility**: Ensure sufficient contrast ratios
5. **Testing**: Test themes across different devices and browsers

## Examples

### Complete Theme Implementation

```tsx
// theme.ts
export const themes = {
  corporate: {
    '--primary': '210 100% 45%',
    '--secondary': '210 15% 20%',
    '--accent': '210 100% 55%',
    '--background': '210 15% 8%',
    '--foreground': '210 15% 95%',
    '--muted': '210 15% 15%',
    '--muted-foreground': '210 15% 70%',
    '--border': '210 15% 18%',
    '--radius': '0.5rem'
  },
  creative: {
    '--primary': '270 100% 60%',
    '--secondary': '320 100% 50%',
    '--accent': '45 100% 55%',
    '--background': '270 15% 5%',
    '--foreground': '270 5% 95%',
    '--muted': '270 15% 10%',
    '--muted-foreground': '270 5% 70%',
    '--border': '270 15% 15%',
    '--radius': '1rem'
  }
}

// Apply theme
function applyTheme(themeName: keyof typeof themes) {
  const theme = themes[themeName]
  const root = document.documentElement
  
  Object.entries(theme).forEach(([property, value]) => {
    root.style.setProperty(property, value)
  })
}
```

### Theme Selector Component

```tsx
import { useState } from 'react'
import { GlassButton, GlassCard } from 'glass-ui-tulio'

function ThemeSelector() {
  const [activeTheme, setActiveTheme] = useState('default')
  
  const themes = [
    { name: 'default', label: 'Default', primary: '217 91% 60%' },
    { name: 'corporate', label: 'Corporate', primary: '210 100% 45%' },
    { name: 'creative', label: 'Creative', primary: '270 100% 60%' },
    { name: 'nature', label: 'Nature', primary: '120 100% 40%' }
  ]
  
  return (
    <GlassCard className="p-6">
      <h3 className="text-lg font-semibold mb-4">Choose Theme</h3>
      <div className="grid gap-3 md:grid-cols-2">
        {themes.map((theme) => (
          <GlassButton
            key={theme.name}
            variant={activeTheme === theme.name ? 'default' : 'outline'}
            onClick={() => {
              setActiveTheme(theme.name)
              applyTheme(theme.name)
            }}
            className="flex items-center gap-2"
          >
            <div 
              className="w-4 h-4 rounded-full"
              style={{ background: `hsl(${theme.primary})` }}
            />
            {theme.label}
          </GlassButton>
        ))}
      </div>
    </GlassCard>
  )
}
```
